def accuracy_score(truth, pred):
    """ Returns accuracy score for input truth and predictions. """
    
    # Ensure that the number of predictions matches number of outcomes
    if len(truth) == len(pred): 
        
        # Calculate and return the accuracy as a percent
        return "Predictions have an accuracy of {:.2f}%.".format((truth == pred).mean()*100)
    
    else:
        return "Number of predictions does not match number of outcomes!"

Concluso

Aps vrias iteraes de explorao e condicionamento nos dados, voc construiu um algoritmo til para prever a sobrevivncia de cada passageiro a bordo do RMS Titanic. A tcnica aplicada neste projeto uma implementao manual de um modelo simples de aprendizado de mquina, a rvore de deciso . Uma rvore de deciso divide um conjunto de dados em grupos menores e menores (chamados ns ), por um recurso por vez. Cada vez que um subconjunto dos dados dividido, nossas previses se tornam mais precisas se cada um dos subgrupos resultantes for mais homogneo (contm rtulos similares) do que antes.

A vantagem de ter um computador faz as coisas para ns que ser mais exaustivo e mais preciso do que a nossa explorao manual acima. Este link fornece outra introduo ao aprendizado de mquina usando uma rvore de deciso.

Uma rvore de deciso apenas um dos muitos modelos que vm da aprendizagem supervisionada . Na aprendizagem supervisionada, tentamos usar recursos dos dados para prever ou modelar as coisas com rtulos de resultados objetivos. Ou seja, cada um dos nossos pontos de dados tem um valor de resultado conhecido, como um rtulo discreto e categrico como "Sobrevivido" ou um valor contnuo numrico, como prever o preo de uma casa.

difcil medir a qualidade de um modelo dado sem quantificar seu desempenho durante o treinamento e teste. Para este projeto, voc ir calcular o coeficiente de determinao, R2, para quantificar o desempenho do seu modelo. O coeficiente de determinao uma estatstica til no campo de anlise de regresso uma vez que descreve o quo "bom" a capacidade do modelo em fazer estimativas.

Os valores para R2 tm um alcance de 0 a 1, que captura a porcentagem da correlao ao quadrado entre a estimativa e o valor atual da varivel alvo. Um modelo R2 de valor 0 sempre falha ao estimar a varivel alvo, enquanto que um modelo R2 de valor 1, estima perfeitamente a varivel alvo. Qualquer valor entre 0 e 1 indica qual a porcentagem da varivel alvo (ao utilizar o modelo) que pode ser explicada pelos atributos. Um modelo pode dar tambm um R2 negativo, que indica que o modelo no melhor do que aquele que estima ingenuamente a mdia da varivel alvo.

from sklearn.metrics import r2_score

def performance_metric(y_true, y_predict):
    """ Calcular e retornar a pontuao de desempenho entre 
        valores reais e estimados baseado na mtrica escolhida. """
    
    score = r2_score(y_true, y_predict)
    
    return score

from sklearn.tree import DecisionTreeRegressor
from sklearn.metrics import make_scorer
from sklearn.model_selection import GridSearchCV

def fit_model(X, y):
    """ Desempenhar busca em matriz sobre o parmetro the 'max_depth' para uma 
        rvore de deciso de regresso treinada nos dados de entrada [X, y]. """
    
    # Gerar conjuntos de validao-cruzada para o treinamento de dados 
    cv_sets = ShuffleSplit(n_splits = 10, test_size = 0.20, random_state = 0)

    regressor = DecisionTreeRegressor()

    params = {'max_depth': np.arange(1, 11)}

    scoring_fnc = make_scorer(performance_metric)

    grid = GridSearchCV(regressor, param_grid=params, scoring=scoring_fnc, cv=cv_sets)

    # Ajustar o objeto de busca em matriz com os dados para calcular o modelo timo
    grid = grid.fit(X, y)

    # Devolver o modelo timo depois de realizar o ajuste dos dados
    return grid.best_estimator_
